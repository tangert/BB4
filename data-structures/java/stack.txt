//We will use a linkedlist to implement the stack.

public class Stack {
    
    //Stack constructor
    public Stack(){
        list = new LinkedList();
    }
    
    public void push(Object item){
        list.add(item);
    }
    
    public Object pop() {
        Object item = list.get(list.size());
        list.remove(list.size());
        return item;
    }
    
    public Object peek() {
        return list.get(list.size());
    }
    
}
