//We will use a linkedlist to implement the queue.

public class Queue {
	
	//Queue constructor
	public Queue() {
		list = new LinkedList();
	}

	public void enqueue(Object item) {
		list.add(item);
	}
    
    public Object dequeue() {
		Object item = list.get(1);
		list.remove(1);
		return item;
	}

	public Object peek() {
		return list.get(1);
	}
}
